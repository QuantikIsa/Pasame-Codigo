class ['a] queue = object (self)
  val mutable front = []
  val mutable back = []
  method push (e : 'a) =
    back <- e :: back
  method top =
    match front, back with
      h::_, _ -> Some h
    | [], [] -> None
    | [], _ ->  front <- List.rev back;
                back  <- [];
                self#top
  method pop =
    match front, back with
      h::t, _ -> front <- t; Some h
    | [], [] -> None
    | [], _ ->  front <- List.rev back;
                back  <- [];
                self#pop
end;;

class ['a] queue' = object (self)
  inherit ['a] queue
  method copy = Oo.copy self
end;;
